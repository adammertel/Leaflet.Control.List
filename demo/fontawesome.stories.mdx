# FontAwesome integration

FontAwesome icons can be used to build the menu.

**creating a list of items**

```js
const items = [
  {
    id: "bolt",
    icon: "fa-bolt",
    active: true,
    places: [
      [53, 13.425],
      [53.63, 15.18]
    ]
  },
  {
    id: "wind",
    icon: "fa-wind",
    active: true,
    places: [
      [51.7, 14.2],
      [53.6, 11.4],
      [50.7, 11.8],
      [49.7, 14.4]
    ]
  },
...
];
```

**initliazation of the L.Control.Select**

```js
L.control
  .select({
    position: "topleft",
    selectedDefault: items.map(i => i.id),
    multi: true,
    iconChecked: '<i class="fa fa-square" />',
    iconUnchecked: '<i class="far fa-square" />',
    items: items.map(item => {
      return {
        label: '<i class="fa ' + item.icon + '" /> ' + item.id,
        value: item.id
      };
    }),
    onSelect: function(selectedItemValues) {
      items.forEach(item => {
        item.active = selectedItemValues.includes(item.id);
      });
      redrawMarkers();
    }
  })
  .addTo(map);
```

**redraw marker function**

```js
var redrawMarkers = () => {
  markers.forEach(marker => {
    map.removeLayer(marker);
  });
  markers = [];
  items
    .filter(s => s.active)
    .forEach(group => {
      group.places.forEach(place => {
        const marker = L.marker(place);
        marker.setIcon(
          L.divIcon({
            html:
              '<div class="icon"><i class="fa ' +
              group.icon +
              '" />' +
              "</div>",
            className: "marker-icon",
            iconSize: [40, 40]
          })
        );
        marker.addTo(map);
        markers.push(marker);
      });
    });
};
```
